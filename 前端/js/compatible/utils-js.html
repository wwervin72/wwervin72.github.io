# 1. Element.matches

if (!Element.prototype.matches) {
    Element.prototype.matches = Element.prototype.msMatchesSelector || Element.prototype.webkitMatchesSelector
}

# 2. Element.closest

if (!Element.prototype.closest) {
    Element.prototype.closest = function (selector) {
        var el = this;
        do {
            if (el.matches(selector)) return el
            el = el.parentElement || el.parentNode
        } while (el !== null && el.nodeType === 1);
        return null;
    }
}

# 3. 浏览器全屏

全屏可以传入 DOM 节点，如果 DOM 是 document.documentElement 则效果和按下 F11 是一样的，如果是普通的元素，则会将该元素充满浏览器，并且全屏的方法内部的 this 必须要指向 DOM，在退出全屏的时候，必须用 document 调用

function fullScreenHandler (elem = document.documentElement) {
    let isFullScreen = document.fullScreen || document.mozFullScreen || document.webkitIsFullScreen;
    let requestMethod;
    if (!isFullScreen) {
        requestMethod = elem.requestFullScreen || elem.webkitRequestFullScreen || elem.mozRequestFullScreen || elem.msRequestFullScreen;
        requestMethod.call(elem);
    } else {
        requestMethod = document.exitFullscreen || document.mozCancelFullScreen || document.webkitCancelFullScreen;
        requestMethod.call(document);
    }
};

# 4. 获取 DOM 节点的偏移量

getBoundingClientRect() 方法返回元素的大小及其相对于视口的位置 document.body.scrollTop 跟 document.documentElement.scrollTop 这两个是一个功能，只不过在不同的浏览器下会有一个始终为 0，所以做了以上的兼容性处理

var getOffset = function(el) {
      var scrollTop = el.getBoundingClientRect().top +
        document.body.scrollTop + document.documentElement.scrollTop;
      var scrollLeft =
        el.getBoundingClientRect().left +
        document.body.scrollLeft +
        document.documentElement.scrollLeft;
      return {
            top: scrollTop,
            left: scrollLeft
      }
}

# 5. window.scrollY/scrollX

返回文档在垂直/水平方向已滚动的像素值。

pageYOffset 属性是 scrollY 属性的别名：

window.pageYOffset == window.scrollY; // true
const getScrollPosition = (el = window) => ({
  x: el.pageXOffset !== undefined ? el.pageXOffset : el.scrollLeft,
  y: el.pageYOffset !== undefined ? el.pageYOffset : el.scrollTop
});

为了跨浏览器兼容，请使用 window.pageYOffset 代替 window.scrollY。另外，旧版本IE（<9）两个属性都不支持。但是我们不考虑 ie < 9

# 6. window.scrollTo

用法：

window.scrollTo(x-coord,y-coord )
window.scrollTo(options)

<!--如何平滑滚动到页面顶部-->
const scrollToTop = () => {
  const c = document.documentElement.scrollTop || document.body.scrollTop;
  if (c > 0) {
    window.requestAnimationFrame(scrollToTop);
    window.scrollTo(0, c - c / 8);
  }
}

参数说明

  1. x-coord 是文档中的横轴坐标。
  2. y-coord 是文档中的纵轴坐标。
  3. options 是一个包含三个属性的对象:
  4. options.top 等同于 y-coord
  5. options.left 等同于 x-coord
  6. options.behavior 类型String,表示滚动行为,支持参数 smooth(平滑滚动),instant(瞬间滚动),默认值auto,实测效果等同于instant

该方法和 window.scroll 是相同的

window.scrollTo( 0, 1000 );
// 设置滚动行为改为平滑的滚动
window.scrollTo({ 
    top: 1000, 
    behavior: "smooth" 
});

# 7. 获取缩放比例（电脑缩放比例与网页缩放比例处理后的值）

function getZoomRatio() {
	let ratio = 0
	let screen = window.screen
	let ua = navigator.userAgent.toLowerCase()

	if (window.devicePixelRatio !== undefined) {
		ratio = window.devicePixelRatio
	} else if (~ua.indexOf('msie')) {
	    <!--兼容 ie 10及以下版本浏览器-->
		if (screen.deviceXDPI && screen.logicalXDPI) {
			ratio = screen.deviceXDPI / screen.logicalXDPI
		}
	} else if (window.outerWidth !== undefined && window.innerWidth !== undefined) {
		ratio = window.outerWidth / window.innerWidth
	}

	if (ratio) {
		ratio = Math.round(ratio * 100)
	}
	return ratio
}
