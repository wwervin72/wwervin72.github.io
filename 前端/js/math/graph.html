# 已知三点 point1 端点1、junction 连接点、point2 端点2

/**
 * 求两条线的夹角
 * [point1, junction]线段长度 Math.sqrt(Math.pow(x1 - px, 2) + Math.pow(y1 - py, 2))
 * @param {*} point1 端点1 [x1, y1]
 * @param {*} junction 连接点 [px, py]
 * @param {*} point2 端点2 [x2, y2]
 */
export function calcJunctionAngle(point1, junction, point2) {
	let { x: x1, y: y1 } = point1
	let { x: px, y: py } = junction
	let { x: x2, y: y2 } = point2
	// 分别求出起点与中心的连线, 终点与中心的连线的向量坐标表示
	let line1 = {
		x: x1 - px,
		y: y1 - py
	}
	let line2 = {
		x: x2 - px,
		y: y2 - py
	}
	let direction
	// 两条连线求叉乘 如果叉乘为负 表示逆时针旋转
	if (line1.x * line2.y - line1.y * line2.x < 0) {
		direction = -1
	} else {
		direction = 1
	}
	// 端点1到连接点的线段长度
	let point1JunctionLen = Math.sqrt(Math.pow(x1 - px, 2) + Math.pow(y1 - py, 2))
	// 端点1到端点2的长度
	let point1Point2Len = Math.sqrt(Math.pow(x1 - x2, 2) + Math.pow(y1 - y2, 2))
	// 端点2到连接点的线段长度
	let junctionPoint2Len = Math.sqrt(Math.pow(px - x2, 2) + Math.pow(py - y2, 2))
	// 根据 余弦定理 求出夹角的 cos 值
	let cosJuctionAngle = (Math.pow(point1JunctionLen, 2) + Math.pow(junctionPoint2Len, 2) - Math.pow(point1Point2Len, 2)) / (2 * point1JunctionLen * junctionPoint2Len)
	// 再根据夹角的方向，求出角度值
	return Math.round(Math.acos(cosJuctionAngle) * 180 / Math.PI) * direction + 360
}

# 已知两点 求直线方程式

/**
 * 已知直线上的两点P1(X1,Y1) P2(X2,Y2)， P1 P2两点不重合。则直线的一般式方程AX + BY + C = 0中，A B C分别等于：
 * A = Y2 - Y1
 * B = X1 - X2
 * C = X2*Y1 - X1*Y2
 * @param {*} p1 线条上一点
 * @param {*} p2 线条上一点
 */
export function calcLineEquation (p1, p2) {
	return {
		a: p2[1] - p1[1],
		b: p1[0] - p2[0],
		c: p2[0] * p1[1] - p1[0] * p2[1]
	}
}
// a、b、c 求出后，即可求出直线上其余点的坐标

# 求平面两点之间的距离

export function calcDistance(pointA, pointB) {
    let x = 'x', y = 'y'
    if (pointA instanceof Array) x = 0, y = 1
    let dx = Number(pointA[x]) - Number(pointB[x])
    let dy = Number(pointA[y]) - Number(pointB[y])
    return Math.sqrt(dx * dx + dy * dy)
}
console.log(calcDistance([1, 1], [2, 3])); // 2.23606797749979
