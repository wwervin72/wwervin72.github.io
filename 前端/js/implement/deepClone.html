# 1. clone function

# 1.1 使用正则表达式，提取出 function 的参数和函数体

函数分为一般函数和箭头函数，这两个有以下区别：

  1. 箭头函数的 this 指向父级作用域。而普通函数的 this 则指向当前调用该函数的对象。
  2. 箭头函数没有 prototype 属性，而普通函数的 prototype 则是指向其构造函数的原型对象
<!--提取函数参数和函数体的正则表达式-->
const bodyReg = /(?<={)(.|\n)+(?=})/m;
const paramReg = /(?<=\().+(?=\)\s+{)/;
# 1.2 clone 函数

通过是否存在 prototype 来判断当前是普通函数还是箭头函数，箭头函数直接通过 eval(fn.toString()) 来返回一个新的箭头函数，而普通函数则需要使用正则表达式提取出函数参数和函数体，在使用 new Function ([arg1[, arg2[, ...argN]],] functionBody) 构造函数重新构造一个新的函数

const bodyReg = /(?<={)(.|\n)+(?=})/m;
const paramReg = /(?<=\().+(?=\)\s+{)/;
function cloneFn(fn) {
    const stringFn = fn.toString();
    if (fn.prototype) {
        const params = paramReg.exec(stringFn);
        const body = bodyReg.exec(stringFn);
        if (body) {
            if (params) {
                const paramArr = params[0].split(',')
                return new Function(...paramArr, body[0]);
            } else {
                return new Function(body[0]);
            }
        } else {
            return null;
        }
    } else {
        return eval(stringFn)
    }
}

# 2. shallow clone 浅拷贝

js 中类型分为基本类型和引用类型，我们在实现浅拷贝的时候，如果当前是基本类型，则直接拷贝基本类型的值，如果是引用类型，则直接拷贝该对象的内存地址，这样修改了其中的一个对象，也就会影响到另外拷贝的对象

function shallowClone (target) {
    <!--判断当前是对象还是数组-->
    let cloneTarget = Array.isArray(target) ? [] : {}
    for (let key in target) {
        cloneTarget[key] = target[key]
    }
    return cloneTarget;
}

# 3. 使用递归来简易的拷贝一个对象

function simpleClone (target) {
    if (typeof target === 'object') {
        let cloneTarget = Array.isArray(target) ? [] : {}
        for (let key in target) {
            cloneTarget[key] = simpleClone(target[key])
        }
        return cloneTarget;
    } else {
        return target
    }
}

# 4. 解决递归调用里的循环引用的问题

我们需要将当前对象和clone后的对象，通过 key/value 的形式存储起来，但是常用的 Object/Array 只支持使用 string/number 作为 Key 值，还好有 es6+ 提供了 Map，可以支持任意类型的数据作为 Key。但是我们这里不使用 Map，而使用 weakMap

map 和 weakMap 的区别

  1. 是 WeakMap 只接受对象作为键名（null除外），不接受其他类型的值作为键名。Map 可以接受任意类型的数据作为 key
  2. WeakMap的键名所指向的对象，不计入垃圾回收机制。WeakMap 的 key 所引用的对象都是弱引用，即垃圾回收机制不将该引用考虑在内，因此只要该引用对象的其他引用被清除，垃圾回收机制就会释放该对象所占用的内存，也就是说，一旦不再需要，WeakMap 里面的键名对象和所对应的键值对会自动消失，不用手动删除引用。

这里有一点需要注意的是，weakmap 的 key 才是弱引用的， value 依然是正常引用

const map = new WeakMap()
map.set(1,2) // Uncaught TypeError: Invalid value used as weak map key

let key = { name : 'ervin'}
var val = {a: 123}
const target = new WeakMap();
target.set(key, val);
val = null;
console.log(target.get(key)) // {a: 123}
key = null
console.log(target.get(key)) // undefined

function simpleClone (target, map = new WeakMap()) {
    if (typeof target === 'object') {
        let cloneTarget = Array.isArray(target) ? [] : {}
        let existInMap = map.get(target)
        if (existInMap) {
            return existInMap
        }
        map.set(target, cloneTarget)
        for (let key in target) {
            cloneTarget[key] = simpleClone(target[key], map)
        }
        return cloneTarget;
    } else {
        return target
    }
}

# 5. 处理其他类型的数据

# 5.1 获取数据类型
<!--通过 toString 方法获取对象的类型-->
function getType(target) {
    return Object.prototype.toString.call(target);
}
<!--可继续遍历类型-->
const mapTag = '[object Map]';
const setTag = '[object Set]';
const arrayTag = '[object Array]';
const objectTag = '[object Object]';
<!--不可继续遍历类型-->
const boolTag = '[object Boolean]';
const dateTag = '[object Date]';
const errorTag = '[object Error]';
const numberTag = '[object Number]';
const regexpTag = '[object RegExp]';
const stringTag = '[object String]';
const symbolTag = '[object Symbol]';
# 5.2 可继续遍历的类型
function getInit(target) {
    const Ctor = target.constructor;
    return new Ctor();
}

function getType (target) {
    retrun Object.prototype.toString.call(target)
}

function isObject (target) {
    const typeOf = typeof target
    return target !== null && (typeOf === "object" || typeOf === "function")
}

const mapTag = '[object Map]';
const setTag = '[object Set]';
const arrayTag = '[object Array]';
const objectTag = '[object Object]';
const deepTag = [mapTag, setTag, arrayTag, objectTag]

function deepClone (target, map = new WeakMap()) {
    <!--克隆基本类型-->
    if (!isObject(target)) {
        retrun target
    }
    const type = getType(target)
    const cloneTarget
    if (deepTag.includes(type)) {
        cloneTarget = getInit(target)
    }
    const existInMap = map.get(target)
    <!--防止循环引用-->
    if (existInMap) {
        return existInMap
    }
    map.set(target, cloneTarget)
    <!--克隆 set-->
    if (type === setTag) {
        target.forEach(val => {
            cloneTarget.add(deepClone(val, map)
        })
        return cloneTarget
    }
     // 克隆map
    if (type === mapTag) {
        target.forEach((val, key) => {
            cloneTarget.set(key, deepClone(val, map));
        });
        return cloneTarget;
    }
    // 克隆对象和数组
    const keys = type === arrayTag ? null : Object.keys(target);
    forEach(keys || target, (value, key) => {
        if (keys) {
            key = value;
        }
        cloneTarget[key] = deepClone(target[key], map);
    });
    return cloneTarget;
}
# 5.3 不可继续遍历的数据类型

Bool、Number、String、Date、Error这几种类型我们都可以直接用构造函数和原始数据创建一个新对象

function cloneOtherType(targe, type) {
    const Ctor = targe.constructor;
    switch (type) {
        case boolTag:
        case numberTag:
        case stringTag:
        case errorTag:
        case dateTag:
            return new Ctor(targe);
        case regexpTag:
            return cloneReg(targe);
        case symbolTag:
            return cloneSymbol(targe);
        default:
            return null;
    }
}
function cloneSymbol(targe) {
    return Object(Symbol.prototype.valueOf.call(targe));
}
function cloneReg(targe) {
    const reFlags = /\w*$/;
    const result = new targe.constructor(targe.source, reFlags.exec(targe));
    result.lastIndex = targe.lastIndex;
    return result
}
