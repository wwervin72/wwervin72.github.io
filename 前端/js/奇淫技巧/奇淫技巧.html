# 常用技巧

# 1.结构解析

# 1.1 用来交换变量的

通常情况下，当交换两个变量的值时，需要创建一个中间变量，来做临时接收，这样看起来也比较繁琐。使用结构赋值就可以很方便的完成。

<!--以前的做法-->
let a = 1, b = 2, temp;
temp = a;
a = b;
b = temp;
console.log(a, b) // 2, 1

<!--使用结构赋值-->
let a = 1, b = 2;
[b, a] = [a, b];
console.log(a, b) // 2, 1
<!--并且结构赋值还可以用于交换多个变量的值-->
let a = 1, b= 2, c= 3;
[c, a, b] = [a, b, c];
console.log(a, b, c) // 2, 3, 1

# 1.2 读取一些数组的特殊变量，如果不存在则使用默认值

<!--通常情况下，如果我们不严格区分 null undefined false 0 等，可以使用 || 操作符来实现-->
var a = [undefined]
var b = a[0] || 'a'
console.log(b)

<!--如果想要获取数组里的 null false 0 等值，则可以使用结构解析默认值的方式，这种方式只要元素值不全等于 undefined 则不会使用默认值-->
const a = [undefined, false, 0, null];
const [b = 'a', c, d, e] = a;
console.log(b, c, d, e) // white false 0 null
<!--如果获取某个具体的元素，可以把前面的元素通过","跳过-->

# 1.2 截取数组

<!--可以使用传统的方式，数组 api：slice substring 等-->
<!--这种方式只能截取数组的指定下标到结尾，不能截取中间字符串，该方式可以截取多个子字符串，并且不会修改原数组的值-->
var a = [1,2,3,4]
var [, , ...b] = a;
console.log(b) // [3, 4]

# 1.3 拷贝对象一些指定属性之外的属性

<!--一般方法-->
var a = {a:1, b: 2, c:3}
var b = Object.assign({}, a)
delete b.a
delete b.b
console.log(b) // {c: 3}

<!--使用结构解析-->
var a = {a1:1, b: 2, c:3}
var {a1, c, ...b} = a
console.log(b) // {c: 3

# 1.4 解构可迭代器对象

<!--实现一个可迭代对象-->
let i = 5;
let a = {
    next () {
        i--
        if (i < 0) {
            return {done: true}
        }
        return {done: false, value: i}
    },
    [Symbol.iterator] () {
        return this
    }
}
<!--使用解构解析-->
<!--注意这个迭代器只能迭代一次，如果前面使用 for of 迭代过 a 了，则后面输出的都是 undefined-->
let [, , b] = a;
console.log(b) // 2

# 1.5 获取动态属性的值

function greet(obj, dynamicProp) {
    <!--如果不存在 nameProp 属性，则返回默认值，该默认值只会匹配 undefined -->
    const { [dynamicProp]: defaultValue = 'unknown' } = obj;
    return defaultValue;
}
greet({a: 123, b: 456, c: undefined}, 'c') // unknown
greet({a: 123, b: 456, c: null}, 'c') // null
greet({a: 123, b: 456, c: 0}, 'd') // unknown

# 4. 扁平化数组

let result = []
function deepFlat(arr) {
    arr.reduce((acc, el) => {
        if (Array.isArray(el)) {
            deepFlat(el)
        } else {
            acc.push(el)
        }
        return acc
    }, result)
}
var arr = [1,[2,3,[4,[5,6]]]]
deepFlat(arr)
console.log(result); // [ 1, 2, 3, 4, 5, 6 ]
<!--也可以使用 es6 的方法-->
Array.prototype.flat.call(arr, Infinity)

# 5. 将 list 转成成 tree

function treeParse(list, id = null, link = 'parentId', sort = 'sort') {
    let arr = []
    return list.filter(el => {
        if (el[link] !== id) {
            arr.push(el) 
            return false
        } else {
            return true
        }
    })
    .sort((a, b) => a[sort] - b[sort])
    .map(el => ({
        ...el,
        children: treeParse(arr, el.id)
    }))
}

const list = [
    { id: 1, parentId: null, sort: 0 },
    { id: 2, parentId: 1, sort: 2 },
    { id: 3, parentId: 1, sort: 1 },

]

# 12.平滑滚动至顶部

function scrollTop() {
    const c = document.documentElement.scrollTop || document.body.scrollTop
    if (c > 0) {
        window.requestAnimationFrame(scrollToTop)
        window.scrollTo(0, c - c / 8)
    }
}

# 13. 批量顺序执行多个 Promise

function runPromisesInSeries (ps) {
    ps.reduce((p, next) => p.then(next), Promise.resolve());
}
const delay = d =>
  new Promise(r =>
    setTimeout(() => {
      console.log(d);
      r();
    }, d)
  );

runPromisesInSeries([() => delay(1000), () => delay(2000), () => delay(3000)]);

# 16. 计算文本大小

/**
 * @param String text 要计算的文本
 * @param styleDom Element 文本样式继承的对象
 */
function calculateTextWidth(text, styleDom) {
    const canvas = document.createElement('canvas');
    const ctx = canvas.getContext('2d');
    ctx.font = window.getComputedStyle(styleDom)['font'];
    return ctx.measureText(text).width;
}
calculateTextWidth('这是一个字符串', document.body);
