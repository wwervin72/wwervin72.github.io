# 类型

# infer

表示在 extends 条件语句中待推断的类型变量

举个例子：

// 这里的infer p是待推断类型，可以是任意类型，由创建类型时候传入的参数决定
type ParamType<T> = T extends (arg: infer P) => any ? P : T;

interface User {
  name: string;
  age: number;
}

type Func = (user: User) => void;
// ParamType<Func> 传入的方法 Func ；类型是可以赋值给 (arg: infer P) => any 类型的，所以这里 P 就是传入方法的参数 User 类型，最后表达式将这个类型返回
type Param = ParamType<Func>; // Param = User
// string 类型不能赋值给类型 (arg: infer P) => any，所以这里直接返回传入的类型 string
type AA = ParamType<string>; // string

# 应用场景一：提取返回值或者参数类型

提取参数类型即可参照上面给出的例子

提取返回值类型，举个例子：

type ReturnType<T> = T extends (...args: any[]) => infer P ? P : any;
interface User {
  name: string;
  age: number;
}
type Func = () => User;
type Test = ReturnType<Func>; // Test = User

# 应用场景二：提取构造函数参数或者实例的类型

举个例子：

// 构造函数可以表示如下
type c = new (...args: any[]) => any;
// 获取参数类型，通常情况下这里获取的类型是元祖
type getParameters<T extends c> = T extends new (...args: infer P) => any
  ? P
  : never;
type getInstance<T extends c> = T extends new (...args: any[]) => infer P ? P : any;
class TestClass {
  constructor(public name: string, public age: number) {}
}

// Params 是元祖类型
type Params = cc<typeof TestClass>; // [string, number]

type Instance = ii<typeof TestClass>; // TestClass

# 应用场景三：将元祖类型转换为联合类型

举个例子:

// 因为元祖类型是可以赋值给数组类型的，所以依靠 infer 可以将元祖类型转换为联合类型
type getUnionType<T> = T extends Array<infer P> ? P : never;
// Params 为上面获取的构造函数参数类型
type res = getUnionType<Params>; // string | number

# never 底部类型

never 类型为空类型和底部类型。never 类型的变量无法被赋值，与其他类型求交集为自身，求并集不参与运算。

举个例子:

T | never // 结果为T
T & never // 结果为never

never 在实际中的运用

# 应用一：找出指定类型

interface DemoInterface{
    a: string,
    b: number,
    c: (n: number) => number,
    d: (n: number) => number,
    e: (c: string) => string,
}
type getNumFnProps<T, condition> = {
    [k in keyof T]: T[k] extends condition ? k : never
}[keyof T]

// "b" | "c" | "d"
type numFnProps = getNumFnProps<DemoInterface, number | ((n: number) => number)>

const a: numFnProps = 'c';

运算过程如下

// 开始，解析interface
{
    a: string,
    b: number,
    c: (n: number) => number,
    d: (n: number) => number,
    e: (c: string) => string,
}
// type getNumFnProps<T, condition> 分为下面两部进行
// 第一步映射类型
{
    a: never
    b: 'b'
    c: 'c',
    d: 'd',
    e: never
}
// 第二步 [keyof T] 索引取值
// never 将会被过滤掉
never | 'b' | 'c' | 'd' | 'never'

# 应用二：防御性编程

interface Foo {   type: 'foo' } 
interface Bar {   type: 'bar' } 
type All = Foo | Bar

function handleValue(val: All) {
    switch (val.type) {
      case 'foo':
        // 这里 val 被收窄为 Foo
        break
      case 'bar':
        // val 在这里是 Bar
        break
      default:
        // val 在这里是 never，如果上面的 case 没有列举完 All 可能的类型，则会走 default 分之，并且将 val 值赋值给 exhaustiveCheck，但是这里是 never 类型，所以这里如果类型没列举完全则会编译报错
        const exhaustiveCheck: never = val
        break
    }
}

# unknown

指不可预先定义的类型，在很多场景下，它可以替代 any 的功能同时保留静态检查的能力

unknow 类型在静态编译的时候不能调用任何方法或属性，比如

var a : unknown = '123';
// 报错
a.split('');
// 需要断言后再吊用方法
(a as string).split('');

应用一：无法确认类型的场景

这种情况需要在代码中去判断不同类型进行不同操作

function resultValueBySome(arg:unknown) { 
  if (typeof arg === 'string') {} 
  else if (typeof val === 'number') {} 
}

# 内置类型

# Pick<T, U>

type Pick<T, K extends keyof T> = { [P in K]: T[P]; }

# Extract<T, U>

T如果能赋值给U，则返回类型T,否则 never

type Extract<T, U> = T extends U ? T : never

# Exclude<T, U>

T如果能赋值给U，则返回类型Never,否则 T

type Exclude<T, U> = T extends U ? never : T
