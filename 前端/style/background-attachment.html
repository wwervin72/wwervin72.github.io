# 介绍

该 css 属性用来决定背景图像的位置是在视口内固定还是随着包含他的区块滚动

该属性有以下三个选项：

  1. scroll
  2. fixed：表示背景相对于视口固定，即使一个元素拥有滚动机制，背景也不会随着内容滚动，并且背景的定位区域是相对于可滚动的区域而不是包含他们的边框
  3. local：表示背景相对于元素的内容固定。如果一个元素拥有滚动机制，背景将会随着元素的内容滚动。

通过一个小荔枝来具体看一下效果：

<style>
  .box {
    height: 100px;
    width: 230px;
    overflow: auto;
    background-color: #f00;
  }
  div {
    box-sizing: border-box;
  }
  .cnt {
    width: 200px;
    height: 300px;
    overflow: auto;
    background: url(https://interactive-examples.mdn.mozilla.net/media/examples/lizard.png)
        right 3rem top 1rem/15rem no-repeat,
      url(https://interactive-examples.mdn.mozilla.net/media/examples/moon.jpg)
        center/10rem;
    background-attachment: local;
  }
  .inner {
    height: 600px;
  }
</style>
<div class="box">
  <div class="cnt">
    <div class="inner"></div>
  </div>
</div>

# 区别

对于 scroll，我们发现 cnt 的内容滚动时，他的背景图片不会滚动，但是它的父元素 box 滚动时，他的背景图就发生滚动了

对于 fixed，我们可以发现 cnt 的背景图片始终固定在界面上，不会因为 cnt 和 box 的内容滚动而滚动

对于 local，cnt 的背景会随着 cnt 和父元素的内容滚动而滚动

除了滚动方式之间的差别外，我们还发现有背景定位区域的差别，举个例子：

<style>
  .box {
    height: 1000px;
    width: 230px;
    overflow: auto;
    background-color: #f00;
  }
  div {
    box-sizing: border-box;
  }
  .cnt {
    width: 200px;
    height: 300px;
    overflow: auto;
    border: 20px solid pink;
    padding-top: 50px;
    background-image: url(https://interactive-examples.mdn.mozilla.net/media/examples/moon.jpg);
    background-position: 0 20px;
    background-attachment: local;
  }
  .inner {
    height: 600px;
  }
</style>
<div class="box">
  <div class="cnt">
    <div class="inner"></div>
  </div>
</div>

从上面的例子，切换不同的 background-attachment 可以看出，背景的绘制区域表现一致，默认只包括内容区域，除去了 边框的范围，但是 padding 的区域仍然会被绘制背景。但是背景的定位区域就不一样了

local、scroll： 这两者的背景定位区域表现一样，都是吧 border 的大小除去了，比如 cnt 有 20px 的边框，background-position 0 20px; 那么背景绘制的顶部距离 cnt 的顶部为 20px + 20px = 40px

fixed: 同样的设置，他的背景绘制顶部距离 cnt 的顶部为 20px

封装的 wave 代码

import /* webpackChunkName: "wave" */ "./wave.scss";

interface IWaveConf {
	target: HTMLElement;
	radius?: number;
	initSize?: number;
	maxSize?: number;
	img?: string;
	waveLen?: number;
}

const defaultInitSize = 30;
const defaultMaxSize = 250;

export class Wave {
	target: HTMLElement;
	radius: number;
	img: string;
	initSize: number;
	maxSize: number;
	waveLen = 6;
	waves: HTMLElement[];
	constructor({
		target,
		radius = 30,
		initSize = defaultInitSize,
		maxSize = defaultMaxSize,
		img = "https://ss3.bdstatic.com/70cFv8Sh_Q1YnxGkpoWK1HF6hhy/it/u=2841247860,2911841437&fm=26&gp=0.jpg",
	}: IWaveConf) {
		this.target = target;
		this.img = img;
		this.initSize = initSize;
		this.maxSize = maxSize;
		this.radius = radius;
		this.waves = [];
		this.createBg();
		this.pushAniStyleSheet();
		this.target.addEventListener("click", this.createWave.bind(this), false);
	}
	createBg() {
		const bg = document.createElement("div");
		bg.style.cssText += `
			width: 100%;
			height: 100%;
			background-image: url('${this.img}');
			background-attachment: fixed;
			background-position: center center;
			background-size: contain;
		`;
		this.target.appendChild(bg);
	}
	createWave(evt: MouseEvent) {
		const waveWrap = document.createElement("div");
		waveWrap.classList.add("wave_position");
		waveWrap.style.cssText += `
				width: ${this.maxSize}px;
				height: ${this.maxSize}px;
				z-index:${this.waves.length + 1};
				top: ${evt.pageY - this.maxSize / 2}px;
				left:${evt.pageX - this.maxSize / 2}px;
			`;
		for (let i = 0; i < this.waveLen; i++) {
			const waveItem = document.createElement("div");
			waveItem.classList.add("wave");
			waveItem.style.cssText += `
				background-image: url('${this.img}');
				top: calc((100% - ${this.initSize}px) / 2);
				left: calc((100% - ${this.initSize}px) / 2);
				width: ${this.initSize}px;
				height: ${this.initSize}px;
				border-radius: ${this.maxSize}px;
			`;
			waveItem.classList.add(`wave_${i}`);
			waveWrap.appendChild(waveItem);
			if (i === this.waveLen - 1) {
				waveItem.addEventListener("animationend", this.removeWave.bind(this), false);
			}
		}
		this.target.appendChild(waveWrap);
		this.waves.push(waveWrap);
	}
	pushAniStyleSheet() {
		const ani = `
				0% {
					top: calc((100% - ${this.initSize}px) / 2);
					left: calc((100% - ${this.initSize}px) / 2);
					width: ${this.initSize}px;
					height: ${this.initSize}px;
					opacity: 1;
				}
				99% {
					opacity: 1;
				}
				100% {
					top: calc((100% - ${this.maxSize}px) / 2);
					left: calc((100% - ${this.maxSize}px) / 2);
					width: ${this.maxSize}px;
					height: ${this.maxSize}px;
					opacity: 0;
				}
			}
		`;
		const prefix = "@keyframes waveAni {";
		const webkitPrefix = "@-webkit-keyframes waveAni {";
		const mozPrefix = "@-moz-keyframes waveAni {";
		const oPrefix = "@-o-keyframes waveAni {";
		let styleSheet = document.styleSheets[document.styleSheets.length - 1];
		styleSheet.insertRule(prefix + ani);
		if ("webkitBoxSizing" in document.body.style) {
			styleSheet.insertRule(webkitPrefix + ani);
		}
		if ("MozBoxSizing" in document.body.style) {
			styleSheet.insertRule(mozPrefix + ani);
		}
		if ("OBoxSizing" in document.body.style) {
			styleSheet.insertRule(oPrefix + ani);
		}
	}
	removeWave(evt: AnimationEvent) {
		let target = evt.target as HTMLElement;
		this.target.removeChild(target?.parentNode!);
	}
}
