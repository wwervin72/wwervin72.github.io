1.1 参数通过 queryString 的方式传递，类似于 url?a=1&b=2&c=3 的形式（表单 form 提交的时候，注意浏览器的地址栏，也是这种方式，因为 form 提交默认的 Content-Type 就是 application/x-www-form-urlencoded），POST 方法同样的设置 Content-Type: application/x-www-form-urlencoded 也是通过这种方式提交。

// 通过调用 r.ParseForm 来处理数据，转换成一个 map，然后通过 r.Form[fieldName] 方式来获取对应字段的数据
err := r.ParseForm()
if err != nil {
    
}
fmt.Println(r.Form[fieldName])

1.2 Content-Type 为 multipart/form-data 的方式，文件上传等可以使用该方式 通过打印 request.body 通常会发现这样的数据

------WebKitFormBoundaryWdDAe6hxfa4nl2Ig
Content-Disposition: form-data; name="userName"
ervin
------WebKitFormBoundaryWdDAe6hxfa4nl2Ig
Content-Disposition: form-data; name="file1"; filename="out.png"
Content-Type: image/png
binary-data

1.2.1 通过 r.ParseMultipartForm 方式解析

// <!--1024 表示读取数据的最大内存大小-->
// <!--注意该方法也会调用 r.ParseForm() 所以也可以直接使用 r.Form[fieldName] 获取指定字段值-->
r.ParseMultipartForm(1024)
for k, v := range r.MultipartForm.Value {
	fmt.Println(k, ": ", v)
}
fmt.Println(r.Form)
// pwd :  [123]
// userName :  [abc]
// map[a:[1] b:[21] pwd:[123] userName:[abc]]

1.2.2 通过 MultipartReader 方式解析

mr, err := r.MultipartReader()
if err != nil {
	fmt.Println("解析失败")
}
form, _ := mr.ReadForm(1024)
for k, v := range form.Value {
	fmt.Println(k, ": ", v)
}
// userName :  [abc]
// pwd :  [123]

1.2.3 通过字节解析

该方式也需要使用 MultipartReader 来解析数据，只是读取数据的方式和上面不一样

mr, err := r.MultipartReader()
if err != nil {
	fmt.Println("解析失败")
}
for {
	p, err := mr.NextPart()
	if err == io.EOF {
		break
	}
	if err != nil {
		fmt.Println("mr.NextPart() err,", err)
		break
	}
	fmt.Println("part header:", p.Header)
	formName := p.FormName()
	fileName := p.FileName()
	if formName != "" && fileName == "" {
	    <!--普通字段-->
		formValue, _ := ioutil.ReadAll(p)
		fmt.Printf("formName:%s,formValue:%s\n", formName, formValue)
	}
	if fileName != "" {
	    <!--上传文件-->
		fileData, _ := ioutil.ReadAll(p)
		fmt.Printf("fileName:%s,fileData:%s\n", fileName, fileData)
	}
}
// part header: map[Content-Disposition:[form-data; name="userName"]]
// formName:userName,formValue:abc
// part header: map[Content-Disposition:[form-data; name="pwd"]]
//formName:pwd,formValue:123

1.2 通过 raw 形式发送的请求体，可以设置多种 Content-Type，处理经常使用的 application/json，外，还有text、text/plain、application/javascript、application/xml、text/xml、text/html 等几种格式

1.2.1 application/json

// <!--将请求体里的 json 读取为 []byte-->
res, _ := ioutil.ReadAll(r.Body)
ubody := &defs.UserCredential{}
// <!--再将 []byte 转换成我们定义好的 struct 数据类型-->
if err := json.Unmarshal(res, ubody); err != nil {
	sendErrorResponse(w, defs.ErrorRequestBodyParseFailed)
