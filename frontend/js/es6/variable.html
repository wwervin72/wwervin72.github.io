# 1. 数组的结构赋值

    let a = 1, b = 2, c = 3;
    let [a, b, c] = [1, 2, 3];
    let [foo, [[bar], baz]] = [1, [[2], 3]];
    // 如果解析不成功则为undefined
    let [aaa] = [];
    let [bbb, ccc] = [1, 2];

# 2. 不完全解构

    let [x, y] = [1, 2, 3];
    x // 1
    y // 2
    
    let [a, [b], d] = [1, [2, 3], 4];
    a // 1
    b // 2
    d // 4

如果等号右边不是数组，或者不是可遍历的结构，那么将会报错

    // 报错
    let [foo] = 1;
    let [foo] = false;
    let [foo] = NaN;
    let [foo] = undefined;
    let [foo] = null;
    let [foo] = {};

# 3. 默认值

    let [foo = true] = [];
    foo // true
    
    let [x, y = 'b'] = ['a']; // x='a', y='b'
    let [x, y = 'b'] = ['a', undefined]; // x='a', y='b'

ES6内部使用严格相等运算符(===)，判断一个位置是否有值。所以如果一个数组成员不严格等于undefined，默认值是不会生效的。

    let [x = 1] = [undefined];
    x // 1
    
    let [x = 1] = [null];
    x // null

默认值可以使用解析结构赋值的其他变量，但该变量必须已经申明

    let [x = 1, y = x] = [];     // x=1; y=1
    let [x = 1, y = x] = [2];    // x=2; y=2
    let [x = 1, y = x] = [1, 2]; // x=1; y=2
    let [x = y, y = 1] = [];     // ReferenceError

# 4. 对象的解构赋值

    let { foo, bar } = { foo: 'foo', bar: 'bar' };
    foo  // foo
    bar  // bar

对象的解构与数组有一个重要的不同。数组的元素是按次序排列的，变量的取值由它的位置决定；而对象的属性没有次序，变量必须与属性同名，才能取到正确的值。

如果变量名与属性名不一致，必须这样写

    var { foo: baz } = { foo: 'aaa', bar: 'bbb' };
    baz // "aaa"
    
    let obj = { first: 'hello', last: 'world' };
    let { first: f, last: l } = obj;
    f // 'hello'
    l // 'world'
    
    // 实际上对象的解构赋值是下面形式的简写
    let {foo: foo, bar: bar} = {foo: 'foo', bar: 'bar'};

这里的P是模式，不是变量了，因此不会被赋值

    let obj = {
        p: [
            'Hello',
            { y: 'World' }
        ]
    };
    
    let { p: [x, { y }] } = obj;
    x // "Hello"
    y // "World"

对象的解构也可以指定默认值

    var {x = 3} = {};
    x // 3
    
    var {x, y = 5} = {x: 1};
    x // 1
    y // 5
    
    var {x:y = 3} = {};
    y // 3
    
    var {x:y = 3} = {x: 5};
    y // 5
    
    var { message: msg = 'Something went wrong' } = {};
    msg // "Something went wrong"

默认值生效的条件是，对象的默认值严格等于undefined

    var {x = 3} = {x: undefined};
    x // 3
    
    var {x = 3} = {x: null};
    x // null

如果解构模式是嵌套对象，并且子对象所在的父属性不存在，将会报错

    // 报错
    let {foo: {bar}} = {baz: 'baz'};

如果要将一个已经声明的变量用于解构赋值，必须非常小心

    let x;
    {x} = {x: 213};  // 报错
    ({x} = {x: 213});  // 正确写法

由于数组的本质是特殊的对象，因此可以对数组进行对象属性解构


    let arr = [1, 2, 3];
    let {0 : first, [arr.length - 1] : last} = arr;
    first // 1
    last // 3

# 5. 字符串解构赋值

    const [a, b, c, d, e] = 'hello';
    a // h
    b // e
    c // l
    d // l
    e // o
    
    let {length: len} = 'hello';
    len // 5

# 6. 数值和布尔值的解构赋值

    let {toString: s} = 123;
    s === Number.prototype.toString  // true
    let {toString: b} = true;
    b === Boolean.prototype.toString // true

解构赋值的规则是，只要等号右边的值不是对象或者数组，都先将其转换为对象，由于undefined与null无法转换为对象，所有对他们进行解构赋值都会报错

    let {prop: x} = undefined;  // 报错
    let {prop: y} = null;  // 报错

# 7. 函数参数的解构赋值

    function add ([x, y]){
        return x + y;
    }
    add([1, 3]);  // 4
    
    [[1,2], [3, 4]].map(([a, b]) => a + b);  //[3, 7]

# 8. 用途

# 8.1 交换变量的值

    let x = 1;
    let y = 2;
    [x, y] = [y, x];
    x // 2
    y // 1

# 8.2 从函数返回多个值

    // 返回一个数组

    function example() {
        return [1, 2, 3];
    }
    let [a, b, c] = example();
    
    // 返回一个对象
    
    function example() {
        return {
            foo: 1,
            bar: 2
        };
    }
    let { foo, bar } = example();

# 8.3 函数参数的定义

    // 参数是一组有次序的值
    function f([x, y, z]) { ... }
    f([1, 2, 3]);
    
    // 参数是一组无次序的值
    function f({x, y, z}) { ... }
    f({z: 3, y: 2, x: 1});

# 8.4 提取json数据

    let jsonData = {
        id: 42,
        status: "OK",
        data: [867, 5309]
    };
    
    let { id, status, data: number } = jsonData;
    
    console.log(id, status, number);
    // 42, "OK", [867, 5309]

# 8.5 函数参数的默认值

    jQuery.ajax = function (url, {
        async = true,
        beforeSend = function () {},
        cache = true,
        complete = function () {},
        crossDomain = false,
        global = true,
        // ... more config
    }) {
        // ... do stuff
    };

# 8.6 遍历Map解构

    var map = new Map();
    map.set('first', 'hello');
    map.set('second', 'world');
    
    for (let [key, value] of map) {
        console.log(key + " is " + value);
    }
    // first is hello
    // second is world
    
    // 获取键名
    for (let [key] of map) {
      // ...
    }
    
    // 获取键值
    for (let [,value] of map) {
      // ...
    }
